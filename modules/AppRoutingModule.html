import { NgModule } from '@angular/core';
import { RouterModule, Routes } from '@angular/router';
import { EditProfileComponent } from './edit-profile/edit-profile.component';
import { EditProfileGuard } from './guards/edit-profile.guard';
import { LoginGuard } from './guards/login.guard';
import { HomeComponent } from './home/home.component';
import { LoginComponent } from './login/login.component';
import { QuizAnswerComponent } from './quiz-answer/quiz-answer.component';
import { QuizInfoComponent } from './quiz-info/quiz-info.component';
import { RegistrationComponent } from './registration/registration.component';
import { BookmarksComponent } from './bookmarks/bookmarks.component';
import { MasterAnswerComponent } from './master-answer/master-answer.component';
import { QuizCreateComponent } from './quiz-create/quiz-create.component';
import { QuizSearchComponent } from './quiz-search/quiz-search.component';


/**
 * Add component and the path at which they will load in the router
 */
const routes: Routes = [
  {path: "home", component: HomeComponent},
  {path: "register", component: RegistrationComponent, canActivate: [LoginGuard]},
  {path: "login", component: LoginComponent, canActivate: [LoginGuard]},
  {path: "edit-profile", component: EditProfileComponent, canActivate: [EditProfileGuard]},
  {path: "answerQuiz", component: QuizAnswerComponent},
  {path: "quiz-info/:id", component:QuizInfoComponent},
  {path: "bookmarks", component: BookmarksComponent, canActivate: [EditProfileGuard]},
  {path: "quiz-answer/:id", component: MasterAnswerComponent},
  {path: "quiz-create", component: QuizCreateComponent, canActivate: [EditProfileGuard]},
  {path: "quiz-search/:id", component: QuizSearchComponent},
  {path: '**', redirectTo: 'home', pathMatch: 'full'}
];


/**
 * Angular generated code to export the Routes defined within this module.
 */
@NgModule({
  imports: [RouterModule.forRoot(routes)],
  exports: [RouterModule]
})
export class AppRoutingModule { }
