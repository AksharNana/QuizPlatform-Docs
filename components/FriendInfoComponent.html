import { Component } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { User } from '../models/user';
import { HistoryService } from '../services/history.service';
import { HistoryArr } from '../models/history';
import { QuizService } from '../services/quiz.service';
import { Quiz } from '../models/quiz';

@Component({
  selector: 'app-friend-info',
  templateUrl: './friend-info.component.html',
  styleUrls: ['./friend-info.component.scss']
})
export class FriendInfoComponent {
  id: any;
  private sub: any;

  /**
   * The constructor is used to inject the ActivatedRoute, HistoryService, and QuizService dependencies.
   * @param route 
   * @param historyService 
   * @param quizService 
   */
  constructor(private route: ActivatedRoute, private historyService:HistoryService, private quizService:QuizService){}

  /**
   * The component class contains several properties, including id to store the friend's ID, 
   * sub to store the subscription, user to store the logged-in user, 
   * User to store the user object, history to store the friend's history, 
   * quiz to store the quizzes, Name to store the name of the selected friend, and Play_times to store the play times of the friend.
   */
  user = sessionStorage.getItem("User");
  User: User = new User;
  history: HistoryArr[] = [];
  quiz: Quiz[]=[];

  //Initialize variables for play times and name of selected friend
  Name : any;

  Play_times: any;

  /**
   * The ngOnInit method is called when the component is created. 
   * It subscribes to the route parameters to get the friend's ID and 
   * fetches the friend's history using the historyService. It also retrieves the logged-in user's object from session storage.
   */
  ngOnInit() {
    this.sub = this.route.params.subscribe(params =>{
      this.id = params['id'];
      if(this.user != null){
        let userOBJ:User = JSON.parse(this.user);
        this.User = userOBJ;
        this.historyService.get_history(this.id).subscribe((res: HistoryArr[])=>{
          this.history = res;
        });
      }
    })
  }
}
