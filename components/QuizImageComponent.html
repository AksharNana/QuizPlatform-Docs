import { Component, Input } from '@angular/core';
import { QuestionService } from '../services/question.service';
import { AnswerService } from '../services/answer.service';
import { ActivatedRoute } from '@angular/router';
import { FormControl, FormGroup } from '@angular/forms';
import { Question } from '../models/question';
import { Answer } from '../models/answer';
import { QuizService } from '../services/quiz.service';
import { User } from '../models/user';
import { Clipboard } from '@angular/cdk/clipboard';
import { History } from '../models/history';
import { HistoryService } from '../services/history.service';
import { Quiz } from '../models/quiz';
@Component({
  selector: 'app-quiz-image',
  templateUrl: './quiz-image.component.html',
  styleUrls: ['./quiz-image.component.scss']
})
export class QuizImageComponent{
  @Input() timeLeft = 30;

  quiz_id: any;
  answer_id: any;
  quizName: any
  private sub: any;
  buttonDispl = "Share Results"
  link: any;
  done: boolean = false;
  rated: boolean = false;
  user = sessionStorage.getItem("User");
  /**
   * Timer
   */
  countdownTimer: any;
  /**
   * Display how much time is left
   */
  timeLeftDisplay: any;
  /**
   * Score
   */
  scoreDisplay: any;

  boolTimesUp: boolean = false;

  answer_form!: FormGroup;

  /**
   * Inject dependencies
   * @param questionService 
   * @param answerService 
   * @param route 
   * @param quizService 
   * @param clipBoard 
   * @param historyService 
   */
  constructor(private questionService: QuestionService, private answerService: AnswerService, private route: ActivatedRoute, private quizService: QuizService,private clipBoard: Clipboard, private historyService:HistoryService) { }

  questions: Question[] = [];

  answers: Answer[] = [];

  rating_form = new FormGroup({
    rating: new FormControl('')
  });

  /**
   * ngOnInit: The ngOnInit lifecycle hook is called when the component is initialized. 
   * It subscribes to the route parameters to get the quiz ID and fetches the quiz details 
   * and questions from the corresponding services. It also sets up a countdown timer that updates the timeLeft property every second.
   */
  ngOnInit() {
    this.sub = this.route.params.subscribe(params => {
      this.quiz_id = params['id'];
      this.quizService.get_quiz_byID(this.quiz_id).subscribe((res: Quiz) =>{
        this.quizName = res.Name;
      });
      this.questionService.get_quiz_questions(this.quiz_id).subscribe((value) => {
        this.questions = value;

        let answer: any = {}

        this.questions.forEach(element => {
          answer[element._id] = new FormControl('')
          this.answerService.get_question_answer(element._id).subscribe((result: Answer[]) => {
            result.forEach(element => {
              this.answers.push(element)
            });
          })

        });

        this.answer_form = new FormGroup(answer);
      });
      // Update the timer every second
      this.countdownTimer = setInterval(() => {
        this.timeLeft--;
        // Stop the timer when time is up
        if (this.timeLeft === 0) {
          clearInterval(this.countdownTimer);
          this.boolTimesUp = true;
          alert("Time is Up!");
          this.checkAnswers();
        }
      }, 1000);
    });

  }

  /**
   * ngOnDestroy: The ngOnDestroy lifecycle hook is called when the component is about to be destroyed. It clears the countdown timer to prevent memory leaks.
   */
  ngOnDestroy(){
    this.stopTimer();
  }

  /**
   * stopTimer: This method is used to stop the countdown timer by clearing the interval.
   */
  stopTimer() {
    clearInterval(this.countdownTimer);
  }


/**
 * checkAnswers: This method is called when the time is up or the user submits their answers. 
 * It stops the timer, checks the user's answers against the correct answers, calculates the score, and displays the result. 
 * It also handles actions like setting the done property to true if the user is logged in, 
 * saving the user's rating for the quiz, updating the user's plays and history, and generating a link to share the results.
 */
  checkAnswers() {
    this.stopTimer();

    var numCorrect = 0;

    this.questions.forEach(element => {
      let answer = this.answer_form.get(element._id)?.value;
      this.answers.forEach(element => {
        if (element._id === answer && element.Correct) {
          numCorrect++;
        }

      });

    });

    this.timeLeftDisplay = "Time Left: " + this.timeLeft + "s";
    this.scoreDisplay = "You scored " + numCorrect + " out of " + this.questions.length;
    if(this.user != null){
      this.done = true;
    }
    let ans = this.rating_form.get("rating")?.value;
    let USer: User;
    if(this.user != null){
      USer =  JSON.parse(this.user!);
      this.scoreDisplay = USer.FirstName + " scored " + numCorrect + " out of " + this.questions.length;
      this.link = this.scoreDisplay + ", play the quiz now: " + "https://quiz-platform-28fcb.web.app/quiz-info/"+this.quiz_id;
      // Add history:
      let history: History = {
        QuizID: this.quiz_id,
        UserID: USer._id,
        Score: numCorrect
      }
      this.historyService.create_history(history).subscribe((res)=>{
        console.log(res);
      });
      //Update user's plays
      this.historyService.update_user_plays(USer.Plays,USer._id).subscribe((res) =>{
        console.log(res);
        USer.Plays = USer.Plays + 1;
        sessionStorage.setItem("User",JSON.stringify(USer));
      });
    }
  }

    /**
     * addRating: This method is called when the user submits a rating for the quiz. 
     * It retrieves the rating value from the form and sends it to the QuizService to add 
     * the rating for the quiz. It also updates the component's rated property to indicate that the user has rated the quiz.
     */
    addRating(){
      let ans = this.rating_form.get("rating")?.value;
      let USer: User;
      if(this.user != null){
        USer =  JSON.parse(this.user!);
        if(ans == null){
          ans = "0"
        }
        this.quizService.addRating(parseInt(ans),this.quiz_id,USer._id).subscribe((res) =>{
          alert("Successfully rated the quiz!");
          console.log(res);
          this.rated = true;
        });
      }
    }

    /**
     * CopyResults: This method is called when the user clicks the "Share Results" button. 
     * It copies the result link to the user's clipboard using the Clipboard service from Angular CDK. 
     * It also updates the buttonDispl property to indicate that the results have been copied.
     */
    CopyResults(){
      this.clipBoard.copy(this.link);
      this.buttonDispl = "Copied Results";
    }
}
