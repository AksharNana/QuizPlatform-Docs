import { Component, OnDestroy, OnInit } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { QuizService } from '../services/quiz.service';
import { Quiz } from '../models/quiz';
import { UserService } from '../services/user.service';

@Component({
  selector: 'app-quiz-info',
  templateUrl: './quiz-info.component.html',
  styleUrls: ['./quiz-info.component.scss']
})
//Overall, this component is responsible for fetching and displaying information about a specific quiz based on the provided
export class QuizInfoComponent implements OnInit, OnDestroy {
  id: any;
  private sub: any;

  /**
   * Constructor: The constructor is used for dependency injection, where instances of services are injected into the component.
   * @param route 
   * @param quizService 
   * @param userService 
   * @param router 
   */
  constructor(private route: ActivatedRoute, private quizService: QuizService, private userService: UserService, private router: Router){}
  quiz!: Quiz;

  Name: any;

  Description: any;

  Rating: any;

  Category: any;

  Total_Played: any;

  Created_By: any;
  Thumbnail: any;
  timeValue = 0;

  /**
   * ngOnInit: The ngOnInit lifecycle hook is called when the component is initialized. 
   * It subscribes to the route parameters to get the quiz ID and fetches the quiz details from the QuizService based on the provided ID. 
   * It assigns the fetched quiz data to the component's properties for display. 
   * Additionally, it retrieves the category and creator information by making additional requests to the QuizService and UserService.
   */
  ngOnInit() {
    this.sub = this.route.params.subscribe(params =>{
      this.id = params['id'];

      this.quizService.get_quiz_byID(this.id).subscribe((value : Quiz) =>{
        this.quiz = value;
        this.Name = value.Name;
        this.Description = value.Description;
        this.Rating = value.Average;
        this.Category = value.Category;
        this.Total_Played = value.Plays;
        this.Created_By = value.CreatorID;
        this.timeValue = value.Timer;
        this.Thumbnail = value.Thumbnail;
        this.quizService.get_category_byID(this.Category).subscribe(res =>{
          this.Category = res;
        });
        this.userService.get_usernameID(this.Created_By).subscribe(res =>{
          this.Created_By = res;
        })
      })
    });
  }

  /**
   * ngOnDestroy: The ngOnDestroy lifecycle hook is called when the component is about to be destroyed. It unsubscribes from the route parameter subscription to prevent memory leaks.
   */
  ngOnDestroy(){
      this.sub.unsubscribe();
  }

  /**
   * navigate: This function is called when the user clicks the "Start Quiz" button. It navigates to the appropriate quiz component based on the quiz's type. The navigation is performed using the Router service from Angular.
   */
  navigate(){
    if(this.quiz.Type==="Text-MCQ"){
      this.router.navigate(['/quiz-answer',this.id]);
    }else if(this.quiz.Type === "Text-Fill-In"){
      this.router.navigate(['/quiz-typein',this.id]);
    }else if(this.quiz.Type === "Image-MCQ"){
      this.router.navigate(['/quiz-image',this.id]);
    }
  }

}
