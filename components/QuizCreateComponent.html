import { Component, OnInit, getModuleFactory } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { CreateQuiz } from '../models/quiz';
import { User } from '../models/user';
import { Quiz } from '../models/quiz';
import { QuizService } from '../services/quiz.service'
import { Category } from '../models/categories';
import { NewQuestion, Question } from '../models/question';
import { NewQuestionAnswer } from '../models/qna';
import { QuestionService } from '../services/question.service';
import { AnswerService } from '../services/answer.service';
import { Answer, NewAnswer } from '../models/answer';
@Component({
  selector: 'app-quiz-create',
  templateUrl: './quiz-create.component.html',
  styleUrls: ['./quiz-create.component.scss']
})
export class QuizCreateComponent implements OnInit {
  constructor(private quizService: QuizService, private questionService: QuestionService, private answerService: AnswerService){}
  user: User = new User;
  quiz: Quiz = new Quiz;
  quizSaved: boolean = false;

  tempQuiz: CreateQuiz = new CreateQuiz;
  categories: Category[] = [];
  questions: NewQuestionAnswer[] = [];

  quizType: string = "";

  ngOnInit(){
    let use = sessionStorage.getItem("User");
    this.user = JSON.parse(use!);
    this.quizService.get_quiz_categories().subscribe((res)=>{
      this.categories = res;
    });
  }

  createQuizForm = new FormGroup({
    quizName: new FormControl('',[
      Validators.required,
    ]),
    quizDescription: new FormControl('', [
      Validators.required
    ]),
    quizCategory: new FormControl('', [
      Validators.required
    ]),
    quizType: new FormControl('',[
      Validators.required
    ]),
    quizTimer: new FormControl('', [
      Validators.required
    ])
  });

  questionForm = new FormGroup({
    question: new FormControl('',[
      Validators.required,
    ]),
    answer: new FormControl('', [
      Validators.required,
    ]),
    truth: new FormControl('false',[
      Validators.required,
    ]),
  });

  questionFormText = new FormGroup({
    question: new FormControl('',[
      Validators.required,
    ]),
    answer: new FormControl('', [
      Validators.required,
    ])
  });
//The method creates a new quiz object based on the user input from a form and saves it to a temporary variable called "tempQuiz"
  createQuiz(){
    this.quizService.get_quiz_categories().subscribe((res: Category[]) =>{
      let qName = this.createQuizForm.get("quizName")?.value!;
      let qDescription = this.createQuizForm.get("quizDescription")?.value!;
      let qCategory = this.createQuizForm.get("quizCategory")?.value!;
      qCategory = res.find(cat => cat.Name === qCategory)?._id!;
      let qType = this.createQuizForm.get("quizType")?.value!;
      let qTimer = Number(this.createQuizForm.get("quizTimer")?.value);
      if(!this.createQuizForm.valid){
        alert("Please complete all the fields!");
      }else{
        let newQuiz: CreateQuiz = {
          Name: qName,
          Description: qDescription,
          CreatorID: this.user._id,
          Category: qCategory,
          Type: qType,
          Plays: 0,
          Average: 0,
          Ordered: true,
          Timer: qTimer,
        }
        this.tempQuiz = newQuiz;
        this.quizSaved = !this.quizSaved;
        alert("Quiz saved, please continue to add questions!");
      }
    })
  }
//method creates a new question object based on the user input from a form and pushes it to an array called "questions".
  createQuestion(){
    if(this.questionForm.valid){
      let question = this.questionForm.get("question")?.value;
      let answer = this.questionForm.get("answer")?.value;
      let correct = this.questionForm.get("truth")?.value;
      let qna: NewQuestionAnswer = {
        QuizID: "ReplaceID",
        Question: question!,
        Answer: answer!,
        Correct: JSON.parse(correct!)
      }
      console.log(qna);
      this.questions.push(qna);
    }else{
      alert("Invalid question and answer, please fill in all the fields!");
    }
  }

  createQuestionText(){
    if(this.questionFormText.valid){
      let question = this.questionFormText.get("question")?.value;
      let answer = this.questionFormText.get("answer")?.value;
      let qna: NewQuestionAnswer = {
        QuizID: "ReplaceID",
        Question: question!,
        Answer: answer!,
        Correct: true
      }
      console.log(qna);
      this.questions.push(qna);
    }else{
      alert("Invalid question and answer, please fill in all the fields!");
    }
  }
//This function is called when the user clicks the "Remove" button next to a question.
  removeQuestion(question: NewQuestionAnswer){
    let index = -1;
    for(let i = 0; i < this.questions.length; i++){
      if(this.questions[i].Question === question.Question && this.questions[i].Answer === question.Answer){
        index = i;
        break;
      }
    }
    this.questions.splice(index,1);
  }
// function is used to publish a quiz
  publishQuiz(){
    if(this.questions.length == 0 || this.tempQuiz.Name === ""){
      alert("Please save quiz and add questions before proceeding!");
    }else{
      this.quizService.create_quiz(this.tempQuiz).subscribe((res) =>{
        if(res['CreatorID']){
          this.quiz = res;
        }
        console.log(JSON.stringify(this.quiz));

        let newQuestions: NewQuestion[] = [];
        for(let i=0; i < this.questions.length; i++){
          let duplicate = false;
          for(let j=0; j < newQuestions.length; j++){
            if(newQuestions[j].Question === this.questions[i].Question){
              duplicate = true;
            }
          }
          if(!duplicate){
            let qnew: NewQuestion = {
              Question : this.questions[i].Question,
              QuizID: this.quiz._id,
              Type: "Text",
            }
            newQuestions.push(qnew);
          }
        }
        this.questionService.create_many_questions(newQuestions).subscribe((res) =>{
          let newQ: Question[] = res;
          let manyAns: NewAnswer[] = [];
          for(let i = 0; i < newQ.length; i++){
            for(let j = 0; j < this.questions.length; j++){
              if(newQ[i].Question === this.questions[j].Question){
                let nAns: NewAnswer = {
                  QuestionID: newQ[i]._id,
                  Answer: this.questions[j].Answer,
                  Correct: this.questions[j].Correct,
                  Type: "Text"
                }
                manyAns.push(nAns);
              }
            }
          }
          this.answerService.create_many_answers(manyAns).subscribe((res) =>{
            alert("Quiz creation completed!");
          });
        });
      });
    }
  }
}
