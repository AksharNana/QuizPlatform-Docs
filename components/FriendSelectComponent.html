import { Component } from '@angular/core';
import { User } from '../models/user';
import { UserService } from '../services/user.service';
import { FRIEND } from '../models/friend';
import { Clipboard } from '@angular/cdk/clipboard';
import { ActivatedRoute } from '@angular/router';
import { HistoryService } from '../services/history.service';
import { HistoryArr } from '../models/history';

@Component({
  selector: 'app-friend-select',
  templateUrl: './friend-select.component.html',
  styleUrls: ['./friend-select.component.scss']
})
export class FriendSelectComponent {

  user!: User;
  hist_id: any;
  buttonDispl = "Copy ID";

  //Get current user
  current_user = sessionStorage.getItem("User");
  friends: FRIEND[] = [];
  
  /**
   * The constructor is used to inject the UserService, Clipboard, ActivatedRoute, and HistoryService dependencies.
   * @param userService 
   * @param clipboard 
   * @param router 
   * @param historyService 
   */
  constructor(private userService: UserService, private clipboard: Clipboard, private router:ActivatedRoute, private historyService: HistoryService) { }

  /**
   * The ngOnInit method is called when the component is created. 
   * It retrieves the current user object from session storage and sets the user property. 
   * It then calls the getFriends method of the UserService to fetch all friends of the 
   * current user and stores them in the friends property. It also subscribes to the route parameters to get the history ID.
   */
  ngOnInit(): void {

    let userOBJ: User;
    if (this.current_user != null) {
      userOBJ = JSON.parse(this.current_user);
      this.user = userOBJ;

      //Get all friends of current user
      this.userService.getFriends(this.user._id).subscribe((res: FRIEND[]) => {
        this.friends = res;
      });

      this.router.params.subscribe((params)=>{
        this.hist_id = params['id'];
      });

    }
  }
  searchedItem: string = "";

/**
 * The search method is used to perform a search for a friend based on the provided value. It updates the searchedItem property with the search value.
 * @param value 
 */
  search(value: string) {
    this.searchedItem = value;
  }

  /**
   * The challengeFriend method is called when a friend is selected to initiate a challenge. 
   * It creates a challenge link with the current user's username and the history ID, copies 
   * the link to the clipboard using the Clipboard service, and displays an alert message confirming that the challenge link has been copied.
   */
  challengeFriend(){
    let link = this.user.Username + " has challenged you! Accept here: https://quiz-platform-28fcb.web.app/challenge/" + this.hist_id;
    this.clipboard.copy(link);
    alert("Challenge copied to clipboard!");
  }

}
