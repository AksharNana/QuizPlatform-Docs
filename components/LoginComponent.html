import { Component } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';

/**
 * Angular Directive
 */
@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.scss']
})
export class LoginComponent {
  /**
     * Create a FormGroup and setup form controls for each field along with Validators
    */
  login_form = new FormGroup({
    email: new FormControl('', [
      Validators.required,
      Validators.email
    ]),
    password: new FormControl('', [
      Validators.required,
      Validators.minLength(8)
    ])
  });

  /**
   * Function that will update current user
  */
  public login() {
    console.warn(this.login_form.value);
    this.login_form.reset();
  }

  /**
   * Returns email form control
   */
  get email() {
    return this.login_form.get("email");
  }
  /**
   * Returns password form control
   */
  get password() {
    return this.login_form.get("password");
  }
}
