import { Component, OnInit, ChangeDetectorRef, ChangeDetectionStrategy, ViewChild, ElementRef} from '@angular/core';
import { Bookmark } from '../models/bookmarks';
import { BookmarkService } from '../services/bookmark.service';
import { QuizService } from '../services/quiz.service';
import { User } from '../models/user';
import { Quiz } from '../models/quiz';

@Component({
  selector: 'app-bookmarks',
  templateUrl: './bookmarks.component.html',
  styleUrls: ['./bookmarks.component.scss'],
})

export class BookmarksComponent implements OnInit{
  bookmarks: Bookmark[] = [];
  quizzes: Quiz[] = [];
  user: User = JSON.parse(window.sessionStorage.getItem("User")!);
  searchedItem: string =  "";
  constructor(private bookmarkService: BookmarkService, private quizService: QuizService){}
  
  ngOnInit(){
    this.bookmarkService.get_bookmarks(this.user._id).subscribe((res: Bookmark[]) =>{
      this.bookmarks = res;
      this.bookmarks.forEach(element => {
        this.quizService.get_quiz_byID(element.QuizID).subscribe((res : Quiz) =>{
          this.quizzes.push(res)
        });
      });
    });
  }

  removeBookmark(quiz : Quiz, idx: number){
    this.bookmarkService.delete_bookmark(this.bookmarks.at(idx)!).subscribe((res) =>{
      if(res['message']){
        this.quizzes.splice(idx,1);
        let index = -1;
        for(let i = 0; i < this.bookmarks.length; i++){
          if(this.bookmarks[i].QuizID == quiz._id){
            index = i;
            break;
          }
        }
        this.bookmarks.splice(index,1);
      }else{
        alert("Delete failed!");
      }
    });
  }

  search(value: string){
    this.searchedItem = value;
  }
  
}
