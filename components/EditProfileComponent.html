import { Component } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { EditUser, User } from '../models/user';
import { UserService } from '../services/user.service';
import { confirmPassword } from '../utilities/password.validator';

/**
 * Angular Directive
 */
@Component({
  selector: 'app-edit-profile',
  templateUrl: './edit-profile.component.html',
  styleUrls: ['./edit-profile.component.scss']
})
export class EditProfileComponent{
  /**
   * Global variable for logged in user
   */
  userObj : User;

  /**
     * Create a FormGroup and setup form controls for each field along with Validators
    */
  update_form = new FormGroup({
    first_name: new FormControl('', [
      Validators.required
    ]),
    surname: new FormControl('', [
      Validators.required
    ]),
    username: new FormControl('', [
      Validators.required
    ]),
    email: new FormControl('', [
      Validators.required,
      Validators.email
    ]),
    password: new FormControl('', [
      Validators.required,
      Validators.minLength(8)
    ]),
    confirm_password: new FormControl('', [
      Validators.required,
    ])
  }, { validators: confirmPassword });

  /**
   * Constructor to inject userService and to populate form with current users information
   */
  constructor(public userService: UserService){
    let user = sessionStorage.getItem("User");
    this.userObj = JSON.parse(user!);
    this.first_name?.setValue(this.userObj.FirstName);
    this.surname?.setValue(this.userObj.Surname);
    this.username?.setValue(this.userObj.Username);
    this.email?.setValue(this.userObj.Email);
    this.password?.setValue(this.userObj.Password);
    this.confirm_password?.setValue(this.userObj.Password);
  }

  /**The updateUser method is responsible for updating the user's profile. 
   * It gathers the updated user information from the form's controls and 
   * makes a call to the edit_user method of the UserService. 
   * The updated user information is sent to the server, and if the update is successful, 
   * the response is logged and the user's session storage is updated with the updated user object.  
   * */
  public updateUser() {
    const userId = this.userObj._id;
    let user: EditUser = {
      FirstName: this.first_name?.value!,
      Surname: this.surname?.value!,
      Username: this.username?.value!,
      Email: this.email?.value!,
      Password: this.password?.value!,
    }
    /**
     * Edit user api call
     */
    this.userService.edit_user(user,userId).subscribe(
      (response : User) => { 
        console.log(response);
        sessionStorage.setItem("User", JSON.stringify(response));
      });
  }

  /**
   * Returns first_name form control
   */
  get first_name() {
    return this.update_form.get("first_name");
  }
  /**
   * Returns surname form control
   */
  get surname() {
    return this.update_form.get("surname");
  }
  /**
   * Returns username form control
   */
  get username() {
    return this.update_form.get("username");
  }
  /**
   * Returns email form control
   */
  get email() {
    return this.update_form.get("email");
  }
  /**
   * Returns password form control
   */
  get password() {
    return this.update_form.get("password");
  }
  /**
   * Returns confirm_password form control
   */
  get confirm_password() {
    return this.update_form.get("confirm_password");
  }
}
