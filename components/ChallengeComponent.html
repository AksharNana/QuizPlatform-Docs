import { Component, Input, OnDestroy, OnInit } from '@angular/core';
import { FormControl, FormGroup } from '@angular/forms';
import { QuestionService } from '../services/question.service';
import { AnswerService } from '../services/answer.service';
import { Question } from '../models/question';
import { Answer } from '../models/answer';
import { ActivatedRoute } from '@angular/router';
import { QuizService } from '../services/quiz.service';
import { User } from '../models/user';
import { History, HistoryArr } from '../models/history';
import { HistoryService } from '../services/history.service';
import { Clipboard } from '@angular/cdk/clipboard';
import { Quiz } from '../models/quiz';
import { FRIEND } from '../models/friend';
import { UserService } from '../services/user.service';

@Component({
  selector: 'app-challenge',
  templateUrl: './challenge.component.html',
  styleUrls: ['./challenge.component.scss']
})
export class ChallengeComponent {
  @Input() timeLeft = 30;

  /**
   * Declare variables required by this component
   */
  quiz_id: any;
  answer_id: any;
  done: boolean = false;
  private sub: any;
  history!: HistoryArr;
  friendname: any;
  winner: any;
  /**
   * Timer
   */
  countdownTimer: any;
  /**
   * Display how much time is left
   */
  timeLeftDisplay: any;
  /**
   * Score
   */
  scoreDisplay: any;

  /**
   * Set the variable for when time is up to false at start
   */
  boolTimesUp: boolean = false;

  /**
   * Declare empty form group, will have answers dynamically added in ngOnInit function
   */
  answer_form!: FormGroup;

  /**
   * The constructor initializes the component with the required services. (Dependency injection)
   * @param userService 
   * @param questionService 
   * @param answerService 
   * @param route 
   * @param quizService 
   * @param historyService 
   * @param clipBoard 
   */
  constructor(private userService: UserService ,private questionService: QuestionService, private answerService: AnswerService, private route: ActivatedRoute, private quizService: QuizService, private historyService: HistoryService, private clipBoard: Clipboard) { }

  /**
   * Questions array
   */
  questions: Question[] = [];

  /**
   * Array of answers
   */
  answers: Answer[] = [];


  /**
   * Gets user from session storage, user=null if the client is not logged in
   */
  user = sessionStorage.getItem("User");

  /**
   * Declare Rating form, once quiz is done, present this form to rate the quiz
   */
  rating_form = new FormGroup({
    rating: new FormControl('')
  });

  quizName: string = "";

/**
 * The ngOnInit method is called when the component is created. 
 * It retrieves the quiz ID from the route parameters and fetches the quiz questions, 
 * answers, and user details using the corresponding services. It also starts a 
 * countdown timer and updates the time left every second.
 */
  ngOnInit() {

    this.sub = this.route.params.subscribe(params => {
      //Get quiz id sent by the parameter and set the variables that will display the quiz questions and values
      let historyID = params['id'];
      this.historyService.get_history_by_id(historyID).subscribe((result: HistoryArr) =>{
        this.history = result;
        this.quizService.get_quiz_byID(result.QuizID).subscribe((res: Quiz) => {
          this.quizName = res.Name;
          this.questionService.get_quiz_questions(result.QuizID).subscribe((value) => {
            this.questions = value;
            
            let answer: any = {}
  
            //Get user's answers to the quiz's questions
            this.questions.forEach(element => {
              answer[element._id] = new FormControl('')
              this.answerService.get_question_answer(element._id).subscribe((result: Answer[]) => {
                result.forEach(element => {
                  this.answers.push(element)
                });
              })
              
            });
            
            this.answer_form = new FormGroup(answer);
            this.userService.get_usernameID(this.history.UserID).subscribe((res)=>{
              this.friendname = res;
            });
          });
        });
      });
      // Update the timer every second
      this.countdownTimer = setInterval(() => {
        this.timeLeft--;
        // Stop the timer when time is up
        if (this.timeLeft === 0) {
          clearInterval(this.countdownTimer);
          this.boolTimesUp = true;
          alert("Time is Up!");
          this.checkAnswers();
        }
      }, 1000);
    });

  }

  /**
   * The ngOnDestroy method is called when the component is destroyed. 
   * It stops the timer by clearing the interval.
   */
  ngOnDestroy() {
    this.stopTimer();
  }

  /**
   * Function to stop countdown timer
   */
  stopTimer() {
    clearInterval(this.countdownTimer);
  }


  /**
   * The checkAnswers method is called when the time is up or when the user submits their answers. 
   * It stops the timer, checks the user's answers against the correct answers, calculates the score, 
   * and displays the results.
   */
  checkAnswers() {
    this.stopTimer();

    var numCorrect = 0;

    //Check if answers are correct
    this.questions.forEach(element => {
      let answer = this.answer_form.get(element._id)?.value;
      this.answers.forEach(element => {
        if (element._id === answer && element.Correct) {
          //Increase the value for all correct answers
          numCorrect++;
        }

      });

    });

    //Display time and the score the user got
    this.timeLeftDisplay = "Time Left: " + this.timeLeft + "s";
      this.scoreDisplay = "You scored " + numCorrect + "/" + this.questions.length + " and your friend scored: " + this.history.Score + "/" + this.questions.length;
    if(numCorrect < this.history.Score){
      this.winner = this.friendname + " won! Better luck next time!";
    }else if(numCorrect > this.history.Score){
      this.winner = "You won! Congratulations!";
    }else{
      this.winner = "You drew with your friend! A worthy battle!";
    }

    this.done = true;

    let ans = this.rating_form.get("rating")?.value;
    let USer: User;
    if (this.user != null) {
      /**
       * If the user is logged in, the score and history are updated and stored in the session storage.
       */
      USer = JSON.parse(this.user!);
      this.scoreDisplay = USer.FirstName + " scored " + numCorrect + "/" + this.questions.length + " and your friend scored: " + this.history.Score + "/" + this.questions.length;
      // Add history:
      let history: History = {
        QuizID: this.quiz_id,
        UserID: USer._id,
        Score: numCorrect
      }
      this.historyService.create_history(history).subscribe((res) => {
        console.log(res);
      });
      //Update user's plays
      this.historyService.update_user_plays(USer.Plays, USer._id).subscribe((res) => {
        console.log(res);
        USer.Plays = USer.Plays + 1;
        sessionStorage.setItem("User", JSON.stringify(USer));
      });
    }
  }
}
