import {  Component, OnInit } from '@angular/core';
import { QuizService } from '../services/quiz.service';
import { Category } from '../models/categories';
import { Quiz } from '../models/quiz';
import { BookmarkService } from '../services/bookmark.service';
import { User } from '../models/user';
import { AddBookmark } from '../models/bookmarks';
import { Router } from '@angular/router';
import { UserService } from '../services/user.service';

@Component({
  selector: 'app-home',
  templateUrl: './home.component.html',
  styleUrls: ['./home.component.scss']
})
//This component represents the home page of the quiz application.
export class HomeComponent implements OnInit{

  /**
   * The constructor of the component injects several services: QuizService, BookmarkService, Router, and UserService.
   * @param quizService 
   * @param bookmarkService 
   * @param router 
   * @param userService 
   */
  constructor(private quizService: QuizService, private bookmarkService: BookmarkService, private router:Router, private userService:UserService){}
  
  user = sessionStorage.getItem("User");

  catagories: Category[] = [];
  quizzes: Quiz[] = [];
  selectedCategory: string = "";
  selectedCatID: string =  "";

  /**
   * In the ngOnInit method, the component makes API calls to retrieve the quiz categories and all quizzes. 
   * The quiz categories are assigned to the categories property, and the quizzes are assigned to the quizzes property. 
   * Additionally, for each quiz, the component retrieves the username of the quiz creator using the userService, and assigns it to the CreatorID property of the quiz.
   */
  ngOnInit(){
    this.quizService.get_quiz_categories().subscribe((res) =>{
      this.catagories = res;
    });

    this.quizService.getAllQuiz().subscribe((res) =>{
      this.quizzes = res;
      this.quizzes.forEach(quiz => {
        this.userService.get_usernameID(quiz.CreatorID).subscribe(res =>{
          quiz.CreatorID = res.toString();
        });
      });
    });
    
  }
 /**
  * The createBookmark method creates a bookmark for the specified quiz and user if the user is logged in. 
  * It checks if the user is logged in by verifying the user variable. 
  * If the user is logged in, it creates a bookmark object and calls the add_bookmark method of the bookmarkService to add the bookmark. 
  * An alert is displayed based on the success or failure of the bookmark creation.
  * @param quiz 
  */
  createBookmark(quiz: Quiz){
    let userOBJ : User;
    if(this.user != null){
      userOBJ = JSON.parse(this.user);
      let bookmark: AddBookmark = {
        QuizID: quiz._id,
        UserID: userOBJ._id
      };
      this.bookmarkService.add_bookmark(bookmark).subscribe((res) =>{
        if(res['QuizID'] && res['UserID']){
          alert("Bookmark added!");
        }else{
          alert("Bookmark failed to add!");
        }
      });
    }else{
      alert("You need to be logged in to do that!");
    }
  }

  /**
   * The updateValue method updates the selectedCategory and selectedCatID properties based on the provided value. 
   * If the value is "Categories", the selectedCategory is set to an empty string, otherwise, it 
   * finds the corresponding category object and assigns its ID to the selectedCatID.
   * @param value 
   */
  updateValue(value: string){
    this.selectedCategory = value;
    if(value === "Categories"){
      this.selectedCatID = "";
    }else{
      let res = this.catagories.find(val=> val.Name === value);
      this.selectedCatID = res?._id!;
    }
  }

  /**
   * The PlayRandomQuiz method selects a random quiz from the quizzes array and 
   * navigates to the quiz answer page using the Router service. 
   * The URL is constructed based on the selected quiz's ID.
   */
  PlayRandomQuiz(){
    let quiz = this.quizzes[Math.floor(Math.random()*this.quizzes.length)]
    console.log(JSON.stringify(quiz))
    let url='/quiz-answer/'+quiz._id.toString()
    this.router.navigateByUrl(url)
  }

  /**
   * The NavigateQuiz method navigates to the quiz information page for the 
   * specified quiz using the Router service. The URL is constructed based on the selected quiz's ID.
   * @param quiz 
   */
  NavigateQuiz(quiz: Quiz){
    this.router.navigate(['/quiz-info',quiz._id]);
  }
}
