import { Component, Input, OnDestroy, OnInit } from '@angular/core';
import { FormBuilder, FormControl, FormGroup, Validators } from '@angular/forms';
import { QuestionService } from '../services/question.service';
import { AnswerService } from '../services/answer.service';
import { Question } from '../models/question';
import { Answer } from '../models/answer';
import { ActivatedRoute } from '@angular/router';

@Component({
  selector: 'app-quiz-typein',
  templateUrl: './quiz-typein.component.html',
  styleUrls: ['./quiz-typein.component.scss']
})
export class QuizTypeinComponent {
    @Input() timeLeft = 30;

  quiz_id: any;
  answer_id: any;

  private sub: any;


  /**
   * Timer
   */
  countdownTimer: any;
  /**
   * Display how much time is left
   */
  timeLeftDisplay: any;
  /**
   * Score
   */
  scoreDisplay: any;

  boolTimesUp: boolean = false;

  answer_form!: FormGroup;

  constructor(private questionService: QuestionService, private answerService: AnswerService, private route: ActivatedRoute) { }

  questions: Question[] = [];

  answers: Answer[] = [];

  /**
   * Run these functions when the component is initialised
  */
  ngOnInit() {
    this.sub = this.route.params.subscribe(params => {
      this.quiz_id = params['id'];

      this.questionService.get_quiz_questions(this.quiz_id).subscribe((value) => {
        this.questions = value;

        let answer: any = {}

        this.questions.forEach(element => {
          answer[element._id] = new FormControl('')
          this.answerService.get_question_answer(element._id).subscribe((result: Answer[]) => {
            result.forEach(element => {
              this.answers.push(element)
            });
          })

        });

        this.answer_form = new FormGroup(answer);
      });
      // Update the timer every second
      this.countdownTimer = setInterval(() => {
        this.timeLeft--;
        // Stop the timer when time is up
        if (this.timeLeft === 0) {
          clearInterval(this.countdownTimer);
          this.boolTimesUp = true;
          alert("Time is Up!");
          this.checkAnswers();
        }
      }, 1000);
    });

  }

  ngOnDestroy(){
    this.stopTimer();
  }

  /**
   * Function to stop timer
   */
  stopTimer() {
    clearInterval(this.countdownTimer);
  }


  /**
   * Function to check answers
   */
  checkAnswers() {
    this.stopTimer();

    var numCorrect = 0;

    this.questions.forEach(element => {
      let answer = this.answer_form.get(element._id)?.value;
      this.answers.forEach(element => {
        if (element._id === answer && element.Correct) {
          numCorrect++;
        }
      });

    });

    this.timeLeftDisplay = "Time Left: " + this.timeLeft + "s";
    this.scoreDisplay = "You scored " + numCorrect + " out of " + this.questions.length;
  }
}
