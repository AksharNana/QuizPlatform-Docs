import { Component, OnInit } from '@angular/core';
import { QuizService } from '../services/quiz.service';
import { ActivatedRoute } from '@angular/router';
import { Quiz } from '../models/quiz';

@Component({
  selector: 'app-master-answer',
  templateUrl: './master-answer.component.html',
  styleUrls: ['./master-answer.component.scss']
})

 // This component represents the page for answering a quiz.
 // Displays the quiz and its questions to the user.
 
export class MasterAnswerComponent implements OnInit{
  constructor(private quizService: QuizService, private route: ActivatedRoute){}
  quizID: string = ""
  quiz: Quiz = new Quiz;
  quizType = "Text";
  quizTime = 10;
  ngOnInit(){
    this.route.params.subscribe(params =>{
      this.quizID = params['id'];
    });
    this.quizService.get_quiz_byID(this.quizID).subscribe((res) =>{
      this.quiz = res;
      this.quizType = "Text";
      this.quizType = this.quiz.Type;
      this.quizTime = this.quiz.Timer;
    });
    console.log("Quiz time: " + this.quiz.Timer);
  }

}
