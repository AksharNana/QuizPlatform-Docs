import { Component, OnInit } from '@angular/core';
import { LeaderboardService } from '../services/leaderboard.service';
import { User } from '../models/user';

@Component({
  selector: 'app-leaderboard',
  templateUrl: './leaderboard.component.html',
  styleUrls: ['./leaderboard.component.scss']
})
export class LeaderboardComponent implements OnInit {
  
  users: User[] = [];
  score: number = 0;

  /**
   * The constructor of the component injects the LeaderboardService, 
   * which is responsible for retrieving leaderboard data.
   * @param leaderboardService 
   */
  constructor(private leaderboardService: LeaderboardService){};
  
  /**
   * In the ngOnInit method, the component makes an API call to the getLeaderboard method of the 
   * leaderboardService to retrieve the leaderboard data. The returned data is assigned to the users property.
   */
  ngOnInit() {
    this.leaderboardService.getLeaderboard().subscribe((res)=>{
      this.users = res;
    }
    )
  };

  /**
   * The increment method is a simple function that increments the score property by 1.
   */
  increment(){
    this.score +=1;
  }
}
