import { Component, OnInit } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';

/**
 * Angular Directive
 */
@Component({
  selector: 'app-quiz-answer',
  templateUrl: './quiz-answer.component.html',
  styleUrls: ['./quiz-answer.component.scss']
})

export class QuizAnswerComponent implements OnInit {
  /**
   * Timer
   */
  countdownTimer: any;
  /**
   * Display how much time is left
   */
  timeLeftDisplay: any;
  /**
   * Score
   */
  scoreDisplay: any;
  /**
   * How much time if left
   */
  timeLeft = 60; // Total time in seconds

  /**
   * Run these functions when the component is initialised
   */
  ngOnInit() {
    // Update the timer every second
    this.countdownTimer = setInterval(() => {
      this.timeLeft--;
      // Stop the timer when time is up
      if (this.timeLeft === 0) {
        clearInterval(this.countdownTimer);
        //					this.checkAnswers();
      }
    }, 1000);
  }

  /**
   * Function to stop timer
   */
  stopTimer() {
    clearInterval(this.countdownTimer);
  }

  answer_form = new FormGroup({
    q1: new FormControl('',[
      Validators.required
    ]),
    q2: new FormControl('',[
      Validators.required
    ]),
    q3: new FormControl('',[
      Validators.required
    ]),
  });

  get q1() {
    return this.answer_form.get("q1");
  }
  /**
   * Returns surname form control
   */
  get q2() {
    return this.answer_form.get("q2");
  }
  /**
   * Returns username form control
   */
  get q3() {
    return this.answer_form.get("q3");
  }

  /**
   * Function to check answers
   */
  checkAnswers() {
    console.log(this.answer_form.value);
    this.stopTimer();

    const q1 = this.answer_form.get("q1")?.value;
    const q2 = this.answer_form.get("q2")?.value;
    const q3 = this.answer_form.get("q3")?.value;

    var correctAnswers = ["paris", "russia", "tokyo"];
    var userAnswers = [q1, q2, q3];
    var numCorrect = 0;

    for (var i = 0; i < correctAnswers.length; i++) {
      if (userAnswers[i] === correctAnswers[i]) {
        numCorrect++;
      }
    }

    this.timeLeftDisplay = "Time Left: " + this.timeLeft + "s";
    this.scoreDisplay = "You scored " + numCorrect + " out of " + correctAnswers.length;
  }

}
