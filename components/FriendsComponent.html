import { Component, OnInit } from '@angular/core';
import { User } from '../models/user';
import { UserService } from '../services/user.service';
import { FRIEND } from '../models/friend';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { Clipboard } from '@angular/cdk/clipboard';


@Component({
  selector: 'app-friends',
  templateUrl: './friends.component.html',
  styleUrls: ['./friends.component.scss']
})
export class FriendsComponent implements OnInit {
  user!: User;

  buttonDispl="Copy ID";

  current_user = sessionStorage.getItem("User");
  friends: FRIEND[] = [];

  ngOnInit(): void {

    let userOBJ: User;
    if (this.current_user != null) {
      userOBJ = JSON.parse(this.current_user);
      this.user = userOBJ;
      this.userService.getFriends(this.user._id).subscribe((res: FRIEND[])=>{
        res.forEach(element => {
          this.userService.get_usernameID(element.FriendID).subscribe((response) =>{ 
            element.FriendID = String(response)
          });
        });
        this.friends = res;
      });
    }
  }
  searchedItem: string =  "";
  constructor(private userService: UserService, private clipboard: Clipboard){}

  friend_form = new FormGroup({
    friend_id: new FormControl('', [
      Validators.required,
    ])
  });

  addFriend(){
    let friendid = this.friend_form.get("friend_id")?.value;
    if(friendid === this.user._id){
      alert("You can't add yourself as a friend!");
    }else if(friendid?.length != 24){
      alert("Invalid friend id!");
    }else{
      this.userService.create_friends(friendid,this.user._id).subscribe((res) =>{
        if(res['friend_exists'] == true){
          alert("Friend already exists!");
        }else{
          this.friends.push(res);
          console.log(this.friends);
        }
      });
    }
  }

  removeFriend(friend : FRIEND, idx: number){
    this.userService.delete_friend(friend._id).subscribe((res) =>{
      if(res["message"]==true){
        this.friends.splice(idx,1);
      }else{
        alert("Delete friend failed!");
        console.log(res);
      }
    })
  }

  copyID(){
    this.clipboard.copy(this.user._id);
    this.buttonDispl = "Copied";
  }

  search(value: string){
    this.searchedItem = value;
  }

}
