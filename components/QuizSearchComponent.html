import { Component, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { QuizService } from '../services/quiz.service';
import { Quiz } from '../models/quiz';
import { User } from '../models/user';
import { AddBookmark } from '../models/bookmarks';
import { BookmarkService } from '../services/bookmark.service';
import { UserService } from '../services/user.service';

@Component({
  selector: 'app-quiz-search',
  templateUrl: './quiz-search.component.html',
  styleUrls: ['./quiz-search.component.scss']
})
export class QuizSearchComponent implements OnInit{

  /**
   * constructor(): The constructor function is responsible for initializing the QuizSearchComponent class. 
   * It injects the required dependencies, such as ActivatedRoute, QuizService, BookmarkService, and UserService, using dependency injection.
   * @param route 
   * @param quizService 
   * @param bookmarkService 
   * @param userService 
   */
  constructor(private route: ActivatedRoute, private quizService: QuizService, private bookmarkService: BookmarkService, private userService: UserService){}

  id = "";
  quizzes: Quiz[]=[];

  user = sessionStorage.getItem("User");

  /**
   * ngOnInit(): This is a lifecycle hook that is called after the component has been initialized. 
   * In this function, the component retrieves the route parameters using ActivatedRoute and subscribes 
   * to the params observable. It retrieves the value of the id parameter from the route and assigns it to the id property of the component. 
   * It then calls the search_quiz method of QuizService to fetch the quizzes based on the provided id. The fetched quizzes are stored in the quizzes property of the component. 
   * For each quiz, it calls the get_usernameID method of UserService to fetch the username associated with the creator ID of the quiz. 
   * The fetched username is assigned to the CreatorID property of the quiz.
   */
  ngOnInit(): void {
      this.route.params.subscribe(params => {
        this.id = params['id'];
        this.quizService.search_quiz(this.id).subscribe(res => {
          this.quizzes = res;
          this.quizzes.forEach(quiz => {
            this.userService.get_usernameID(quiz.CreatorID).subscribe(res =>{
              quiz.CreatorID = res.toString();
            });
          });
        });
      });
  }

  /**
   * createBookmark(quiz: Quiz): This function is called when the user wants to create a bookmark for a quiz. 
   * It takes a quiz parameter of type Quiz, which represents the quiz for which the bookmark is to be created. 
   * Inside the function, it checks if a user is logged in by checking the user property obtained from the sessionStorage. 
   * If a user is logged in, it parses the user object from sessionStorage and creates a bookmark object of type AddBookmark, 
   * which contains the QuizID and UserID. It then calls the add_bookmark method of BookmarkService and subscribes to the returned observable. 
   * If the bookmark is successfully added, it displays an alert message. 
   * Otherwise, it displays an alert message indicating the failure to add the bookmark. 
   * If no user is logged in, it displays an alert message stating that the user needs to be logged in to perform the bookmark action.
   * @param quiz 
   */
  createBookmark(quiz: Quiz){
    let userOBJ : User;
    if(this.user != null){
      userOBJ = JSON.parse(this.user);
      let bookmark: AddBookmark = {
        QuizID: quiz._id,
        UserID: userOBJ._id
      };
      this.bookmarkService.add_bookmark(bookmark).subscribe((res) =>{
        if(res['QuizID'] && res['UserID']){
          alert("Bookmark added!");
        }else{
          alert("Bookmark failed to add!");
        }
      });
    }else{
      alert("You need to be logged in to do that!");
    }
  }
}
