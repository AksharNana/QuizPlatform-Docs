import { Injectable } from '@angular/core';
import { HttpClient, HttpErrorResponse, HttpHeaders } from '@angular/common/http';
import { NewQuestion, Question, UpdateQuestion } from '../models/question';
import { catchError, Observable, throwError } from 'rxjs';
import { Category } from '../models/categories';
import { handleError } from '../utilities/errorhandler';

/**
 * Default angular directive
 */
@Injectable({
  providedIn: 'root'
})
export class QuestionService {

  /**
   * Constructor to inject http client
   * @param http
   */
  constructor(private http: HttpClient) { }

  /**
   * Base url for question calls
   */
  base_url = "https://clumsy-apron-ray.cyclic.app/api/question";

  /**
   * Creates a question for quiz
   * @param question 
   * @returns Question
   */
  create_question(question: NewQuestion): Observable<any> {
    let nquestion = JSON.stringify(question);
    return this.http.post<Question>(this.base_url + "/createQuestion", nquestion, {
      headers: new HttpHeaders({
        'Content-Type': 'application/json'
      })
    }).pipe(
      catchError(handleError)
    );
  }

  /**
   * Create many questions
   * @param questions 
   * @returns Question[]
   */
  create_many_questions(questions: NewQuestion[]) : Observable<any>{
    let nquestions = JSON.stringify(questions);
    return this.http.post<Question[]>(this.base_url+"/createManyQuestions",nquestions,{
      headers: new HttpHeaders({
        'Content-Type':'application/json'
      })
    }).pipe(
      catchError(handleError)
    );
  }

  /**
   * Gets all questions for all quizzes
   * @returns Question[]
   */
  getAllQuestions() : Observable<any[]>{
    return this.http.get<Question[]>(this.base_url+"/getAllQuestions").pipe(
      catchError(handleError)
    );
  }

  /**
   * Get all questions belonging to a specifc quiz
   * @param id 
   * @returns Question[]
   */
  get_quiz_questions(id: String) : Observable<any[]>{
    return this.http.get<Question[]>(this.base_url+"/getQuestions/"+id).pipe(
      catchError(handleError)
    );
  }
  
  /**
   * Updates question values
   * @param id 
   * @param question 
   * @returns Question
   */
  update_question(id:String, question:UpdateQuestion) : Observable<any>{
    let nquestion = JSON.stringify(question);
    return this.http.patch<Question>(this.base_url+"/updateQuestion/"+id, nquestion,{
      headers: new HttpHeaders({
        'Content-Type': 'application/json'
      })
    });
  }

  /**
   * Removes question from database and associated answers.
   * @param question 
   * @returns 
   */
  delete_question(question: Question) : Observable<any>{
    return this.http.delete(this.base_url+"/deleteQuestion/"+question._id).pipe(
      catchError(handleError)
    );
  }

  create_image(file: FormData): Observable<any>{
    return this.http.post("https://clumsy-apron-ray.cyclic.app/api/image/upload",file);
  }
}
