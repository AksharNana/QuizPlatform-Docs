Base url for answer calls

 
import { HttpClient, HttpErrorResponse, HttpHeaders } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { catchError, Observable, throwError } from 'rxjs';
import { Answer, NewAnswer, UpdateAnswer } from '../models/answer';
import { handleError } from '../utilities/errorhandler';

/**
 * Angular Directive
 */
@Injectable({
  providedIn: 'root'
})
export class AnswerService {

  /**
   * Angular Constructor to inject http Client
   * @param http 
   */
  constructor(private http: HttpClient) { }

  /**
   * Base url for answer calls
   */
  base_url = "https://clumsy-apron-ray.cyclic.app/api/answer";

    /**
   * Create Answer request
   */
  create_answer(answer: NewAnswer): Observable<any> {
    let nans = JSON.stringify(answer);
    return this.http.post(this.base_url + "/createAnswer", nans, {
      headers: new HttpHeaders({
        'Content-Type': 'application/json'
      })
    }).pipe(
      catchError(handleError)
    );
  }

  /**
   * Creates many answers
   * @param answerArr 
   * @returns Answer[]
   */
  create_many_answers(answerArr: NewAnswer[]) : Observable<any[]>{
    let nans = JSON.stringify(answerArr);
    return this.http.post<Answer[]>(this.base_url+"/createManyAnswers",nans, {
      headers: new HttpHeaders({
        'Content-Type': 'application/json'
      })
    }).pipe(
      catchError(handleError)
    );
  }
  
  /**
   * Get all answers for all quizzes
   * @returns Answer[]
   */
  getAllAnswers() : Observable<any[]>{
    return this.http.get<Answer[]>(this.base_url+"/getAllAnswers").pipe(
      catchError(handleError)
    );
  }

  /**
   * Get answers to a specific question
   * @param QuestionID 
   * @returns Answer[]
   */
  get_question_answer(QuestionID: String) : Observable<any[]>{
    return this.http.get<Answer[]>(this.base_url+"/getAnswers/"+QuestionID).pipe(
      catchError(handleError)
    );
  }

  /**
   * Updates an answer
   * @param id 
   * @param answer 
   * @returns Answer
   */
  update_answer(id:String, answer:UpdateAnswer) : Observable<any>{
    let nans = JSON.stringify(answer);
    return this.http.patch<Answer>(this.base_url+"/updateAnswer/"+id, nans,{
      headers: new HttpHeaders({
        'Content-Type': 'application/json'
      })
    });
  }
  
  /**
   * Delete the answer from database
   * @param id 
   * @returns any
   */
  delete_answer(id: string) : Observable<any>{
    return this.http.delete(this.base_url+"/deleteAnswer/"+id).pipe(
      catchError(handleError)
    );
  }
  
}
