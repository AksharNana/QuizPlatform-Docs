import { HttpClient, HttpErrorResponse, HttpHeaders } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { catchError, Observable, throwError } from 'rxjs';
import { Category } from '../models/categories';
import { CreateQuiz, Quiz } from '../models/quiz';
import { handleError } from '../utilities/errorhandler';

/**
 * Angular Directive
 */
@Injectable({
  providedIn: 'root'
})
export class QuizService {

  /**
   * Constructor to inject http Client
   * @param http
   */
  constructor(private http: HttpClient) { }

  /**
 * Base url for quiz calls
 */
  base_url = "https://clumsy-apron-ray.cyclic.app/api/quiz";

  /**
   * Creates a new quiz
   * @param newQuiz 
   * @returns Quiz
   */
  create_quiz(newQuiz: CreateQuiz): Observable<any> {
    let nQuiz = JSON.stringify(newQuiz);
    return this.http.post<Quiz>(this.base_url + "/createQuiz", nQuiz,
      {
        headers: new HttpHeaders({
          'Content-Type': 'application/json'
        })
      }).pipe(
        catchError(handleError)
      );
  }

  /**
   * Get all Quizzes from Quiz database
   * @returns Quiz[]
   */
  getAllQuiz(): Observable<any[]> {
    return this.http.get<Quiz[]>(this.base_url + "/getAllQuizzes").pipe(
      catchError(handleError)
    );
  }

  /**
   * Update the values of a particular quiz
   * @param update 
   * @param id 
   * @returns Quiz
   */
  update_quiz(update : Quiz, id: String) : Observable<any>{
    let nupdate = JSON.stringify(update);
    return this.http.patch<Quiz>(this.base_url+"/updateQuiz/"+ id,nupdate, {
      headers: new HttpHeaders({
        'Content-Type': 'application/json'
      })
    }).pipe(
      catchError(handleError)
    );
  }  

  /**
   * Remove a quiz from the database
   * @param id 
   * @returns any
   */
  delete_quiz(id: String): Observable<any> {
    return this.http.delete(this.base_url + "/deleteQuiz/"+id).pipe(
      catchError(handleError)
    );
  }

  /**
   * Returns all Quiz Categories available
   * @returns Category[]
   */
  get_quiz_categories(): Observable<any[]>{
    return this.http.get<Category[]>("https://clumsy-apron-ray.cyclic.app/api/category/getAllCategories").pipe(
      catchError(handleError)
    );
  }
}
