import { HttpClient, HttpErrorResponse, HttpHeaders } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { catchError, Observable } from 'rxjs';

import { CreateUser, EditUser, LoginUser, User } from '../models/user';
import { handleError } from '../utilities/errorhandler';
import { FRIEND } from '../models/friend';

/**
 * Angular Directive
 */
@Injectable({
  providedIn: 'root'
})
export class UserService {

  /**
   * Constructor to inject http Client
   * @param http
   */
  constructor(private http: HttpClient) { }

  /**
   * All user related requests use this base url
   */
  base_url = "https://clumsy-apron-ray.cyclic.app/api/user";
  user : User = new User();

  /**
   * Check if email can be registered or not
   * @param email 
   * @returns True or false value
   */
  check_email_exists(email: string) : Observable<any>{
    return this.http.get(this.base_url+"/emailExists/"+ email).pipe(
      catchError(handleError)
    );
  }

  /**
   * Take user input and create new user
   * @param user 
   * @returns 
   */
  create_user(user: CreateUser) : Observable<any>{
    let userNew = JSON.stringify(user);
    return this.http.post<CreateUser>(this.base_url+"/createUser",userNew,
    {headers: new HttpHeaders({
      'Content-Type': 'application/json'
    })}).pipe(
      catchError(handleError)
    );
  }
  
  /**
   * Take new info and update current user.
   * @param user 
   * @param userID 
   * @returns 
   */
  edit_user(user: EditUser, userID: String) : Observable<any>{
    let userUpdated = JSON.stringify(user);
    return this.http.patch<EditUser>(this.base_url+"/updateUser/"+userID,userUpdated,
    {headers: new HttpHeaders({
      'Content-Type': 'application/json'
    })}).pipe(
      catchError(handleError)
    );
  }

  /**
   * Log the user in
   * @param email 
   * @param password 
   * @returns User
   */
  login(user : LoginUser) : Observable<any>{
    let body = JSON.stringify(user)
    return this.http.post<any>(this.base_url+"/login",body,{
      headers: new HttpHeaders({
        'Content-Type' : 'application/json'
      })
    }).pipe(
      catchError(handleError)
    );
  }

  /**
   * Create friends
   * @param friendArray
   * @param userID
   */
  create_friends(friendID: string, userID : string) {
    let body = JSON.stringify({
      "Username": userID,
      "Friend": friendID
    });
    return this.http.post<any>("https://clumsy-apron-ray.cyclic.app/api/friend/createFriend",body,{
      headers: new HttpHeaders({
        'Content-Type' : 'application/json'
      })
    }).pipe(catchError(handleError));
  }

  /**
   * Get Friends List
   * @param userID
   * @returns Array of friends IDs
   */
  getFriends(userID: string): Observable<any>{
    return this.http.get("https://clumsy-apron-ray.cyclic.app/api/friend/getFriends/"+userID).pipe(
      catchError(handleError)
    );
  }

  /**
   * Remove friends
   * @param friendObjID 
   * @returns Message: true/false
   */
  delete_friend(friendObjID: string): Observable<any>{
    return this.http.delete("https://clumsy-apron-ray.cyclic.app/api/friend/deleteFriend/"+friendObjID).pipe(
      catchError(handleError)
    );
  }

  /**
   * Get username by id
   * @param id 
   * @returns User Name
   */
  get_usernameID(id:string){
    return this.http.get(this.base_url+"/getName/"+id).pipe(
      catchError(handleError)
    );
  }
}
